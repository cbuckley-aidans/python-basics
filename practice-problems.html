# Coin Flip Challenge
# A simple game where the player guesses the outcome of a coin flip

import random

# Display welcome message and game rules
print("COIN FLIP CHALLENGE")
print("I'll flip a coin. Can you guess if it will be heads or tails?")
print("----------------------------------------------------")

# Get the player's guess
player_guess = input("Enter your guess (heads/tails): ").lower()

# Make sure the guess is valid
while player_guess != "heads" and player_guess != "tails":
    print("Invalid guess. Please enter 'heads' or 'tails'.")
    player_guess = input("Enter your guess (heads/tails): ").lower()

# Flip the coin (generate a random result)
print("\nFlipping the coin...")
coin_results = ["heads", "tails"]
coin_flip = random.choice(coin_results)

# Check if the player's guess matches the result
print(f"The coin landed on: {coin_flip}")

# Compare guess with result and display appropriate message
if player_guess == coin_flip:
    print("Congratulations! Your guess was correct. You win!")
else:
    print(f"Sorry! Your guess was incorrect. You guessed {player_guess}, but the coin landed on {coin_flip}.")
